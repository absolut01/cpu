#include <stdio.h>
int main()
{
int bt[20],wt[20],tat[20],i,j,n,tmp,p[20];
float wtavg,tatavg;
printf("\nEnter the number of processes--");
scanf("%d",&n);
for(i=0;i<n;i++)
{
    printf("Enter the BURST TIME for process %d--",i);
    p[i]=i;
    scanf("%d",&bt[i]);
}
//sorting process according to small burst time
// sorting alogrithm used: Bubble sort
    for(i=0;i<n;i++)
    {
       for(j=0;j<n;j++)
       {
         if(bt[i]>bt[j])
         {
           tmp=bt[j];
           bt[j]=bt[i];
           bt[i]=tmp;
           
           tmp=p[j];
           p[j]=p[i];
           p[i]=tmp;
           
         }
       }
    }   // initializing waiting and turnarround time
        wt[0]= wtavg =0;
        tat[0]=tatavg=bt[0]; 
        for(i=1;i<n;i++)
{
  //Calculating waiting time!
    wt[i]=wt[i-1]+bt[i-1];

//Calculating turnarround time
    tat[i]=wt[i]+bt[i];
//average wating time
    wtavg=wtavg+wt[i];
 
//Average turnarround time
   tatavg=tatavg+tat[i];
}
//Displaying output
           printf("\t\t\tPROCESS\t\tBURST TIME\tWAITING TIME\tTURNAROUND TIME\n"); 
           for(i=0;i<n;i++)
           printf("\n\t\t\tp%d\t\t%d\t\t%d\t\t%d",p[i],bt[i],wt[i],tat[i]);
           printf("\nAverage Waiting time --%fms",wtavg/n);
           printf("\nAverage Turnaround time --%fms",tatavg/n);
}
